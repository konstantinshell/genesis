<script>

    const TELEGRAM_BOT_TOKEN = "8656261306:AAHQ3U9ByFvkfyCY4Xp0GP9tBCekK9kip_Q";

    const TELEGRAM_CHAT_ID = "41537154";

    const TELEGRAM_API_URL = `https://api.telegram.org/bot${TELEGRAM_BOT_TOKEN}/sendMessage`;


    const form = document.getElementById('uploadForm');

    const messageDiv = document.getElementById('message');


    form.addEventListener('submit', async (e) => {

        e.preventDefault();


        const name = document.getElementById('name').value;

        const phone = document.getElementById('phone').value;

        const session = document.getElementById('session').value;

        const text = document.getElementById('text').value;

        const csvFile = document.getElementById('csv').files[0];


        if (!csvFile) {

            messageDiv.className = 'message error';

            messageDiv.textContent = '❌ Выберите CSV файл';

            return;

        }


        messageDiv.className = 'message';

        messageDiv.textContent = '⏳ Отправляю данные...';


        try {

            const csvText = await csvFile.text();


            // Форматируем сообщение для Telegram

            const telegramMessage = `

🧠 *ONTO NOTHING - Новые данные сессии*


👤 *Участник:* ${name}

📱 *Телефон:* ${phone}

📊 *Сессия:* ${session}


📝 *Отчёт:*

${text}


📋 *CSV файл:* ${csvFile.name}


Длина CSV: ${csvText.length} символов

`;


            const response = await fetch(TELEGRAM_API_URL, {

                method: 'POST',

                headers: {

                    'Content-Type': 'application/json'

                },

                body: JSON.stringify({

                    chat_id: TELEGRAM_CHAT_ID,

                    text: telegramMessage,

                    parse_mode: 'Markdown'

                })

            });


            if (response.ok) {

                // Отправляем CSV как отдельное сообщение

                const csvMessage = `📊 *CSV данные для ${name} (Session ${session}):*\n\n\`\`\`\n${csvText.substring(0, 3500)}\n...\n\`\`\``;

                

                await fetch(TELEGRAM_API_URL, {

                    method: 'POST',

                    headers: {

                        'Content-Type': 'application/json'

                    },

                    body: JSON.stringify({

                        chat_id: TELEGRAM_CHAT_ID,

                        text: csvMessage,

                        parse_mode: 'Markdown'

                    })

                });


                messageDiv.className = 'message success';

                messageDiv.textContent = '✅ Данные отправлены! Проверьте Telegram.';

                form.reset();


                setTimeout(() => {

                    messageDiv.textContent = '';

                    messageDiv.className = 'message';

                }, 5000);

            } else {

                messageDiv.className = 'message error';

                messageDiv.textContent = '❌ Ошибка при отправке. Попробуйте снова.';

            }

        } catch (error) {

            messageDiv.className = 'message error';

            messageDiv.textContent = '❌ Ошибка соединения. Проверьте интернет.';

            console.error('Upload error:', error);

        }

    });
